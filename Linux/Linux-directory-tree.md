```bash
/
├── bin/          # Essential system binaries
├── boot/         # Boot loader and kernel files
├── dev/          # Device files
├── etc/          # System-wide configuration files
├── home/         # Home directories for users
├── lib/          # Shared libraries
├── lib64/        # 64-bit shared libraries
├── media/        # Mount point for removable media
├── mnt/          # Mount point for temporary mounts
├── opt/          # Optional software packages
├── proc/         # Process information and kernel parameters
├── root/         # Home directory for the root user
├── run/          # Runtime variable data
├── sbin/         # System binaries (usually for root)
├── srv/          # Data for services provided by the system
├── sys/          # Kernel-related information (pseudo filesystem)
├── tmp/          # Temporary files
├── usr/          # Secondary hierarchy for user data
│   ├── bin/      # User binaries
│   ├── include/  # Standard include files
│   ├── lib/      # User libraries
│   ├── local/    # Locally installed software
│   ├── sbin/     # System binaries for non-essential programs
│   ├── share/    # Architecture-independent data files
│   └── src/      # Source code for software
|
└── var/          # Variable data (log files, spool directories, etc.)
    ├── cache/    # Application cache data
    ├── lib/      # Variable state information
    └── log/      # Log files
```
